def get_effective_volume_ratio(signal_type: str, current_time: str) -> float:
    """
    신호타입 × 시간대별 동적 거래량 기준
    """
    # 신호타입별 기본값
    base_vol = {
        "UPPER_LIMIT": 3.0,      # 상한가 근처
        "NEAR_UPPER": 5.0,       # 상한가 인접
        "SURGE": 7.0,            # 급등
        "EXECUTION_SPEED": 5.0,  # 체결속도
        "DEFAULT": 5.0
    }
    
    base = base_vol.get(signal_type, 5.0)
    
    # 시간대별 가중치
    hour = int(current_time.split(":")[0])
    if 9 <= hour < 10:           # 장초반
        time_adj = 1.3
    elif 10 <= hour < 14:        # 장중
        time_adj = 1.0
    elif 14 <= hour < 15:        # 장마감 앞
        time_adj = 0.85
    else:                         # NXT
        time_adj = 0.9
    
    required = max(2.0, min(base * time_adj, 10.0))
    return round(required, 1)

# analyze() 함수 내에서 사용

def confirm_bottom_and_signal(watch: dict, current_price: int, current_data: dict) -> dict:
    """바닥 형성 확인 + 재상승 신호 판정"""
    try:
        result = {
            "is_bottom_confirmed": False,
            "should_send_signal": False,
            "signal_type": None,
            "reason": ""
        }
        
        # 필수조건 (ALL)
        essential_check = {
            "volume": current_data.get("volume_ratio", 0) >= 2.5,
            "strength": current_data.get("signal_strength") in ("강함", "매우강함", "A"),
            "time": (datetime.now() - watch.get("low_price_time", datetime.now())).total_seconds() <= 1800
        }
        
        if not all(essential_check.values()):
            result["reason"] = "필수조건 미충족"
            return result
        
        result["is_bottom_confirmed"] = True
        
        # 가산조건 점수 (2개 이상)
        bonus_score = 0
        
        # ① 외국인/기관 부호 전환
        sentiment = detect_foreign_institution_turnaround(watch["code"])
        if sentiment["foreign_turnaround"] and sentiment["institution_buying"]:
            bonus_score += 3
        
        # ② DART 공시 유효성
        if watch.get("dart_reliability_score", 0) > 0:
            bonus_score += 2
        
        # ③ 호가 유동성 양호
        if check_liquidity(watch["code"], max_spread=300):
            bonus_score += 2
        
        # ④ RSI 30 이하
        rsi = get_rsi_5min(watch["code"])
        if rsi <= 30:
            bonus_score += 2
        
        # ⑤ 거래량 감소 (Wyckoff Spring)
        vol_now = get_current_volume(watch["code"])
        vol_prev_low = watch.get("volume_at_low", vol_now)
        if vol_now < vol_prev_low * 0.7:
            bonus_score += 2
        
        # ⑥ MACD 신호
        macd_histogram = get_macd_histogram(watch["code"])
        if macd_histogram > 0:
            bonus_score += 1
        
        # ⑦ 캔들 패턴
        if detect_hammer_or_bullish_engulfing(watch["code"]):
            bonus_score += 1
        
        # 판정
        if bonus_score >= 3:
            result["should_send_signal"] = True
            result["signal_type"] = "entry_signal"
            result["reason"] = f"바닥 확인 + 재상승 신호 ({bonus_score}점)"
        elif bonus_score >= 0:
            result["should_send_signal"] = True
            result["signal_type"] = "watch_alert"
            result["reason"] = f"바닥 형성 확인, 재상승 대기중 ({bonus_score}점)"
        
        return result
    
    except Exception as e:
        return {
            "is_bottom_confirmed": False,
            "should_send_signal": False,
            "signal_type": None,
            "reason": f"오류: {str(e)}"
        }

# check_entry_watch()에서 사용
if log_key in _entry_watch:
    watch = _entry_watch[log_key]
    if price <= entry:
        bottom_check = confirm_bottom_and_signal(watch, price, current_data)
        
        if bottom_check["is_bottom_confirmed"]:
            if bottom_check["signal_type"] == "entry_signal":
                send_alert(f"✅ {stock_name}\n바닥 확인 + 재상승 신호\n{bottom_check['reason']}")
            elif bottom_check["signal_type"] == "watch_alert":
                send_alert(f"🔍 {stock_name}\n바닥 형성 확인\n{bottom_check['reason']}")
        else:
            _entry_watch.pop(log_key)
